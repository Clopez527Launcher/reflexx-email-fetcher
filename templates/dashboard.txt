    <script>
        async function fetchLogs() {
            const response = await fetch('/get_logs');
            const logs = await response.json();
            const logsTable = document.getElementById("logs-table-body");

            logsTable.innerHTML = ""; // Clear previous logs

            logs.forEach(log => {
                const row = `<tr>
                    <td>${log.timestamp}</td>
                    <td>${log.user_id}</td>
                    <td>${log.mouse_distance.toFixed(2)} px</td>
                    <td>${log.keystrokes}</td>
                    <td>${log.mouse_clicks}</td>
                    <td>${log.idle_count}</td>
                </tr>`;
                logsTable.innerHTML += row;
            });
        }

        setInterval(fetchLogs, 5000); // Refresh logs every 5 seconds
        window.onload = fetchLogs; // Load logs on page load
    </script>
